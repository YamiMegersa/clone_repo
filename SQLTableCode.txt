-- 1. Province Table
CREATE TABLE Province (
    ProvinceID INT PRIMARY KEY AUTO_INCREMENT,
    ProvinceName VARCHAR(100) NOT NULL
);

-- 2. Municipality Table
CREATE TABLE Municipality (
    MunicipalityID INT PRIMARY KEY AUTO_INCREMENT,
    ProvinceID INT NOT NULL,
    MunicipalityName VARCHAR(100) NOT NULL,
    FOREIGN KEY (ProvinceID) REFERENCES Province(ProvinceID)
);

-- 3. Ward Table
CREATE TABLE Ward (
    WardID INT PRIMARY KEY,
    MunicipalityID INT NOT NULL,
    WardCouncillor VARCHAR(150),
    FOREIGN KEY (MunicipalityID) REFERENCES Municipality(MunicipalityID)
);

-- 4. Residents Table
CREATE TABLE Residents (
    ResidentID INT PRIMARY KEY AUTO_INCREMENT,
    Username VARCHAR(50) NOT NULL UNIQUE,
    Email VARCHAR(255) NULL UNIQUE,
    CellphoneNumber VARCHAR(15) NULL UNIQUE,
    BlackListed BOOLEAN DEFAULT FALSE
);

-- 5. Municipal Workers Table
CREATE TABLE MunicipalWorkers (
    EmployeeID INT PRIMARY KEY,
    FirstName VARCHAR(100) NOT NULL,
    LastName VARCHAR(100) NOT NULL,
    Email VARCHAR(255) NULL UNIQUE,
    Cell VARCHAR(15) NULL UNIQUE,
    Blacklisted BOOLEAN DEFAULT FALSE,
    Validated BOOLEAN DEFAULT FALSE,
    ProfilePicture LONGBLOB
);

-- 6. Subscription Table
CREATE TABLE Subscription (
    SubscriptionID INT PRIMARY KEY AUTO_INCREMENT,
    WardID INT NOT NULL,
    ResidentID INT NOT NULL,
    FOREIGN KEY (WardID) REFERENCES Ward(WardID),
    FOREIGN KEY (ResidentID) REFERENCES Residents(ResidentID)
);

-- 7. Reports Table
CREATE TABLE Reports (
    ReportID INT PRIMARY KEY AUTO_INCREMENT,
    WardID INT NOT NULL,
    ResidentID INT NOT NULL,
    Latitude DECIMAL(9,6) NOT NULL,
    Longitude DECIMAL(9,6) NOT NULL,
    Progress VARCHAR(50) NOT NULL DEFAULT 'Pending Allocation',
    Type VARCHAR(100),
    Frequency INT NOT NULL,
    CreatedAt DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    DateFulfilled DATETIME NULL,
    Priority INT NULL,
    FOREIGN KEY (WardID) REFERENCES Ward(WardID),
    FOREIGN KEY (ResidentID) REFERENCES Residents(ResidentID)
);

-- 8. Allocation Table
CREATE TABLE Allocation (
    AllocationID INT PRIMARY KEY AUTO_INCREMENT,
    ReportID INT NOT NULL,
    EmployeeID INT NOT NULL,
    Accepted BOOLEAN DEFAULT FALSE,
    FOREIGN KEY (ReportID) REFERENCES Reports(ReportID),
    FOREIGN KEY (EmployeeID) REFERENCES MunicipalWorkers(EmployeeID)
);

-- 9. Report Images Table
CREATE TABLE ReportImages (
    ImageID INT PRIMARY KEY AUTO_INCREMENT,
    ReportID INT NOT NULL,
    Type VARCHAR(50),
    ImageData LONGBLOB NOT NULL,
    FOREIGN KEY (ReportID) REFERENCES Reports(ReportID)
);

-- 10. Grievance Table
CREATE TABLE Grievance (
    GrievanceID INT PRIMARY KEY AUTO_INCREMENT,
    ResidentID INT NULL,  
    EmployeeID INT NULL,  
    Description TEXT NOT NULL,
    Attachment LONGBLOB,
    Resolved BOOLEAN DEFAULT FALSE,
    CreatedAt DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (ResidentID) REFERENCES Residents(ResidentID),
    FOREIGN KEY (EmployeeID) REFERENCES MunicipalWorkers(EmployeeID)
);
