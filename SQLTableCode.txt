-- 1. Province Table
CREATE TABLE Province (
    ProvinceID INT PRIMARY KEY IDENTITY(1,1),
    ProvinceName VARCHAR(100) NOT NULL
);

-- 2. Municipality Table
CREATE TABLE Municipality (
    MunicipalityID INT PRIMARY KEY IDENTITY(1,1),
    ProvinceID INT NOT NULL,
    MunicipalityName VARCHAR(100) NOT NULL,
    FOREIGN KEY (ProvinceID) REFERENCES Province(ProvinceID)
);

-- 3. Ward Table
CREATE TABLE Ward (
    WardID INT PRIMARY KEY,
    MunicipalityID INT NOT NULL,
    WardCouncillor VARCHAR(150),
    FOREIGN KEY (MunicipalityID) REFERENCES Municipality(MunicipalityID)
);

-- 4. Residents Table
CREATE TABLE Residents (
    ResidentID INT PRIMARY KEY IDENTITY(1,1),
    Username VARCHAR(50) NOT NULL,
    Email VARCHAR(255) NOT NULL UNIQUE,
    CellphoneNumber VARCHAR(15) NOT NULL UNIQUE,
    BlackListed BIT DEFAULT 0
);

-- 5. Municipal Workers Table
CREATE TABLE MunicipalWorkers (
    EmployeeID INT PRIMARY KEY ,
    FirstName VARCHAR(100) NOT NULL,
    LastName VARCHAR(100) NOT NULL,
    Email VARCHAR(255) NOT NULL UNIQUE,
    Cell VARCHAR(15) NOT NULL UNIQUE,
    Blacklisted BIT DEFAULT 0,
    Validated BIT DEFAULT 0,
    ProfilePicture VARBINARY(MAX) -- Azure blob
);

-- 6. Subscription Table (Many-to-Many between Residents and Wards)
CREATE TABLE Subscription (
    SubscriptionID INT PRIMARY KEY IDENTITY(1,1),
    WardID INT NOT NULL,
    ResidentID INT NOT NULL,
    FOREIGN KEY (WardID) REFERENCES Ward(WardID),
    FOREIGN KEY (ResidentID) REFERENCES Residents(ResidentID)
);

-- 7. Reports Table
CREATE TABLE Reports (
    ReportID INT PRIMARY KEY IDENTITY(1,1),
    WardID INT NOT NULL,
    ResidentID INT NOT NULL,
    Lattitude DECIMAL(8,4) NOT NULL,
    Longitude DECIMAL(8,4) NOT NULL,
    Progress VARCHAR(50) NOT NULL DEFAULT 'Pending Allocation',
    Type VARCHAR(100),
    Frequency INT NOT NULL,
    Date DATETIME NOT NULL DEFAULT GETDATE(),
    DateFulfilled DATETIME NULL,
    FOREIGN KEY (WardID) REFERENCES Ward(WardID),
    FOREIGN KEY (ResidentID) REFERENCES Residents(ResidentID)
);

-- 8. Allocation Table
CREATE TABLE Allocation (
    AllocationID INT PRIMARY KEY IDENTITY(1,1),
    ReportID INT NOT NULL,
    EmployeeID INT NOT NULL,
    FOREIGN KEY (ReportID) REFERENCES Reports(ReportID),
    FOREIGN KEY (EmployeeID) REFERENCES MunicipalWorkers(EmployeeID)
);

-- 9. Report Images Table
CREATE TABLE ReportImages (
    ImageID INT PRIMARY KEY IDENTITY(1,1),
    ReportID INT NOT NULL,
    Type VARCHAR(50),
    Image VARBINARY(MAX) NOT NULL,
    FOREIGN KEY (ReportID) REFERENCES Reports(ReportID)
);

-- 10. Consolidated Grievance Table
-- Handles Resident, Municipal Worker, and Anonymous grievances in one place.
CREATE TABLE Grievance (
    GrievanceID INT PRIMARY KEY IDENTITY(1,1),
    ResidentID INT NULL,  
    EmployeeID INT NULL,  
    Description TEXT NOT NULL,
    Attachment VARBINARY(MAX),
    Resolved BIT DEFAULT 0,
    Date DATETIME DEFAULT GETDATE(),
    
    FOREIGN KEY (ResidentID) REFERENCES Residents(ResidentID),
    FOREIGN KEY (EmployeeID) REFERENCES MunicipalWorkers(EmployeeID),
    


);
